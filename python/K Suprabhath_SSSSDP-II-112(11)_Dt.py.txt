# Function to get a valid month number and print the month name
def get_month_name():
    try:
        # Input: Month number (should be an integer)
        month_number = input("Enter month number (1 - 12): ")

        # Check if the input is a valid integer
        month_number = int(month_number)

        # Check if the month number is within the valid range
        if month_number == 1:
            print("January")
        elif month_number == 2:
            print("February")
        elif month_number == 3:
            print("March")
        elif month_number == 4:
            print("April")
        elif month_number == 5:
            print("May")
        elif month_number == 6:
            print("June")
        elif month_number == 7:
            print("July")
        elif month_number == 8:
            print("August")
        elif month_number == 9:
            print("September")
        elif month_number == 10:
            print("October")
        elif month_number == 11:
            print("November")
        elif month_number == 12:
            print("December")
        else:
            print("Input should be between 1 and 12")
    
    except ValueError:
        # Handle non-integer input
        print("Input should be an integer")

# Call the function
get_month_name()
