lst = [10, 20, 15, 12, 18, 15, 19, 14, 15, 14]
x = int(input("Enter the element to be searched: "))

count = 0  # To track how many times x is found

for i in range(len(lst)):
    if lst[i] == x:
        print(x, "is found at index", i)
        count += 1

if count > 0:
    print(x, "is found", count, "times")
else:
    print(x, "is not found")

sum = 0
ctr = 0

try:
    while True:
        val = input("Enter input  (ctrl + z  to  stop)  :  ")
        val = eval(val)
        sum += val
        ctr += 1
except EOFError:
    pass

if ctr > 0:
    avg = sum / ctr
    print("Average :", avg)
else:
    print("No input received.")
