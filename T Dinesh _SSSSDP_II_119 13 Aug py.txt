def distinct_vowels(input_string):
    vowels = 'AEIOU'
    result = "" 
    for char in input_string.upper():
        if char in vowels and char not in result:
            result += char

    return result

input_string = input("Enter a string: ")

output = distinct_vowels(input_string)
print("Distinct vowels in the string:", output)
