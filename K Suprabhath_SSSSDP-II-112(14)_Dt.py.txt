import sys

argv = sys.argv

# Case when no inputs are given
if len(argv) == 1:
    print("Pls send inputs")
    sys.exit()

a = []

# Try converting inputs to float or treat them as strings
only_numbers = True
only_strings = True

for x in argv[1:]:
    try:
        val = float(x)
        a.append(val)
        only_strings = False
    except ValueError:
        a.append(x)
        only_numbers = False

# Handle mixed inputs
if not only_numbers and not only_strings:
    print("Inputs can not be number and string")
    sys.exit()

# Output results
print("What is argv ? --->", argv)
print("What is list 'a' ? --->", a)

# Determine and print largest input
largest = max(a)
print("What is the largest command line input ? --->", largest

# Input string
s = input("Enter a string: ")

# Initialize empty strings for even and odd indexed chars
even = ''
odd = ''

# Loop over all indices
for i in range(len(s)):
    if i % 2 == 0:        # Even index
        even += s[i]
    else:                 # Odd index
        odd += s[i]

# Print results
print("even =", even)
print("odd =", odd)

##11
 # Input string
a = 'A4B3C2$5'

# Output string initialization
out = ''

# Loop through string in steps of 2
for i in range(0, len(a), 2):
    char = a[i]        # character at even index
    count = int(a[i+1])  # number at odd index, convert to int
    out += char * count

print(out)

##12

# Input strings
a = 'HYD'
b = 'VAMSI'

c = ''  # Result string
i = 0

# Loop until reaching the end of the shorter string
while i < len(a) and i < len(b):
    c += a[i] + b[i]
    i += 1

# Append the remaining part of the longer string
if i < len(a):
    c += a[i:]
elif i < len(b):
    c += b[i:]

print(c)

a = 'A4M3Z5D2'

out = ''
for i in range(0, len(a), 2):
    char = a[i]
    num = int(a[i+1])
    
    # Calculate shifted character using ord() and chr()
    shifted_char = chr(ord(char) + num)
    
    out += char + shifted_char
